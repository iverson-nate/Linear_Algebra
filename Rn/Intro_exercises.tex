\begin{exercise}
Compute the following:\\
\begin{inparaenum}[a.)]
\begin{tabular}{llll}
\item $3 \begin{bmatrix*}[r] 1 \\ 0\\ -5\end{bmatrix*}$ \hspace{1cm} &
\item $\begin{bmatrix*}[r]1 \\ 0\\ -5\end{bmatrix*} + \begin{bmatrix}4 \\ 1\\ 7\end{bmatrix}$ \hspace{1cm} & 
\item $3\begin{bmatrix*}[r]1 \\ 0\\ -5\end{bmatrix*} + 2\begin{bmatrix}4 \\ 1\\ 7\end{bmatrix}$ \hspace{1cm} &
\item $a\begin{bmatrix*}[r]1 \\ 0\\ -5\end{bmatrix*} + b\begin{bmatrix}4 \\ 1\\ 7\end{bmatrix}$
\end{tabular}
\end{inparaenum}
\end{exercise}

\begin{exercise}
For $0 \in \mathbb{R}$ and $\vec{v} \in \mathbb{R}^n$ show that $0\vec{v}=\vec{0} \in \mathbb{R}^n$.
\end{exercise}

\begin{exercise}
Show that every vecotor in $\mathbb{R}^n$ can be written as a linear combination of the following vectors: $\begin{bmatrix}1\\ 0 \\ 0\end{bmatrix}, \begin{bmatrix}1\\ 1 \\ 0\end{bmatrix},\begin{bmatrix}1\\ 1 \\ 1\end{bmatrix}$
\end{exercise}
